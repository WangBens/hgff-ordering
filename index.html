function doPost(e) { var result = {}; try { var body = JSON.parse(e.postData.contents); var action = body.action; if (action === 'getMenuAndOptions') { // 【新增】同時取得菜單與客製化選項 result = { success: true, data: getMenuAndOptions() }; } else if (action === 'submitOrder') { // 【新增】接收前端送來的新訂單 result = submitOrder(body.userId, body.lineName, body.lastName, body.phone, body.cart, body.totalPrice); } else if (action === 'verifyAdmin') { result = { success: true, isAdmin: verifyAdmin(body.userId) }; } else if (action === 'getAdminDashboardData') { result = { success: true, data: getAdminDashboardData() }; } else if (action === 'adminTopUp') { result = adminTopUp(body.userId, body.amount, body.note); } else if (action === 'processOrderAction') { result = processOrderAction(body.orderId, body.orderAction); } else { result = { success: false, message: '未知的 API 指令' }; } } catch (error) { result = { success: false, message: '後端執行錯誤: ' + error.toString() }; } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput("凰貴妃茶堂 API 伺服器運作中。"); } // ========================================= // 新增功能：取得菜單與客製化選項 // ========================================= function getMenuAndOptions() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var menuSheet = ss.getSheetByName("菜單"); var optSheet = ss.getSheetByName("加料與客製化"); var menuList = []; var options = { sweetness: [], ice: [], toppings: [] }; // 1. 讀取菜單 if (menuSheet) { var mData = menuSheet.getDataRange().getValues(); for (var i = 1; i < mData.length; i++) { if (mData[i][4] === '上架') { menuList.push({ category: mData[i][0], name: mData[i][1], priceL: Number(mData[i][2]), priceXL: Number(mData[i][3]) }); } } } // 2. 讀取客製化選項 if (optSheet) { var oData = optSheet.getDataRange().getValues(); for (var j = 1; j < oData.length; j++) { if (oData[j][3] === '上架') { var type = oData[j][0]; var optObj = { name: oData[j][1], price: Number(oData[j][2]) }; if (type === '甜度') options.sweetness.push(optObj); else if (type === '冰塊') options.ice.push(optObj); else if (type === '加料') options.toppings.push(optObj); } } } return { menu: menuList, options: options }; } // ========================================= // 新增功能：寫入新訂單到資料庫 // ========================================= function submitOrder(userId, lineName, lastName, phone, cart, totalPrice) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var orderSheet = ss.getSheetByName("訂單紀錄"); if (!orderSheet) return { success: false, message: "找不到訂單紀錄工作表" }; // 產生訂單編號 (利用時間戳記) var orderId = "ORD" + new Date().getTime().toString().slice(-6); var orderTime = new Date(); // 組合顧客稱呼 var customerName = lastName + " (" + phone + ") [" + lineName + "]"; // 將購物車陣列轉換成可讀的文字格式 var orderContentText = ""; for (var i = 0; i < cart.length; i++) { var item = cart[i]; var details = [item.sweetness, item.ice]; if (item.toppings.length > 0) details.push("+" + item.toppings.join("、")); orderContentText += "[" + item.size + "] " + item.name + " x" + item.qty + " (" + details.join(", ") + ")\n"; } // 寫入 Google Sheet // A:訂單編號, B:時間, C:LINE_ID, D:姓名暱稱, E:訂單內容, F:總金額, G:付款方式, H:狀態 orderSheet.appendRow([ orderId, orderTime, userId, customerName, orderContentText.trim(), totalPrice, "現場結帳", // 預設付款方式 "未處理" ]); return { success: true, message: "訂單已成功送出！" }; } // ========================================= // 以下為原本的老闆後台功能，保持不變 // ========================================= function verifyAdmin(userId) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("管理員名單"); if (!sheet) return false; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] === userId) return true; } return false; } function getAdminDashboardData() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var orderSheet = ss.getSheetByName("訂單紀錄"); var orders = []; if (orderSheet) { var orderData = orderSheet.getDataRange().getValues(); for (var i = 1; i < orderData.length; i++) { if (orderData[i][7] === "未處理") { orders.push({ id: orderData[i][0], time: Utilities.formatDate(new Date(orderData[i][1]), "GMT+8", "yyyy/MM/dd HH:mm"), userId: orderData[i][2], userName: orderData[i][3], content: orderData[i][4], total: Number(orderData[i][5]), payMethod: orderData[i][6], status: orderData[i][7] }); } } } var userSheet = ss.getSheetByName("會員與餘額"); var users = []; if (userSheet) { var userData = userSheet.getDataRange().getValues(); for (var j = 1; j < userData.length; j++) { users.push({ userId: userData[j][0], userName: userData[j][1], balance: Number(userData[j][2]) }); } } return { orders: orders, users: users }; } function adminTopUp(targetUserId, amount, note) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var userSheet = ss.getSheetByName("會員與餘額"); var logSheet = ss.getSheetByName("儲值紀錄"); if (!userSheet || !logSheet) return { success: false, message: "資料表不存在" }; amount = Number(amount); if (isNaN(amount) || amount <= 0) return { success: false, message: "金額錯誤" }; var userData = userSheet.getDataRange().getValues(); var userRowIndex = -1; var userName = "未知用戶"; for (var i = 1; i < userData.length; i++) { if (userData[i][0] === targetUserId) { userRowIndex = i + 1; userName = userData[i][1]; break; } } if (userRowIndex === -1) return { success: false, message: "找不到會員" }; var currentBalance = Number(userSheet.getRange(userRowIndex, 3).getValue()); var newBalance = currentBalance + amount; userSheet.getRange(userRowIndex, 3).setValue(newBalance); logSheet.appendRow([new Date(), targetUserId, userName, amount, note || "現場現金儲值"]); return { success: true, newBalance: newBalance, userName: userName }; } function processOrderAction(orderId, action) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var orderSheet = ss.getSheetByName("訂單紀錄"); var userSheet = ss.getSheetByName("會員與餘額"); if (!orderSheet) return { success: false, message: "訂單資料表不存在" }; var orderData = orderSheet.getDataRange().getValues(); var orderRowIndex = -1; var targetOrder = null; for (var i = 1; i < orderData.length; i++) { if (orderData[i][0].toString() === orderId.toString()) { orderRowIndex = i + 1; targetOrder = { userId: orderData[i][2], total: Number(orderData[i][5]), payMethod: orderData[i][6], status: orderData[i][7] }; break; } } if (orderRowIndex === -1) return { success: false, message: "找不到訂單" }; if (targetOrder.status !== "未處理") return { success: false, message: "該訂單已被處理過" }; if (action === 'cancel') { orderSheet.getRange(orderRowIndex, 8).setValue("已取消"); return { success: true, message: "訂單已取消" }; } if (action === 'complete') { if (targetOrder.payMethod === "餘額扣款") { if (!userSheet) return { success: false, message: "會員資料表不存在" }; var userData = userSheet.getDataRange().getValues(); var userRowIndex = -1; var currentBalance = 0; for (var j = 1; j < userData.length; j++) { if (userData[j][0] === targetOrder.userId) { userRowIndex = j + 1; currentBalance = Number(userData[j][2]); break; } } if (userRowIndex === -1) return { success: false, message: "扣款失敗" }; if (currentBalance < targetOrder.total) return { success: false, message: "餘額不足" }; userSheet.getRange(userRowIndex, 3).setValue(currentBalance - targetOrder.total); } orderSheet.getRange(orderRowIndex, 8).setValue("已完成"); return { success: true, message: "結帳完成" }; } return { success: false, message: "未知指令" }; }
